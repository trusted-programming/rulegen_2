 case 0 :
    Console.WriteLine ("i is zero.");
    goto case 1;

    Console.WriteLine ("i is one.");

case 2 :
    Console.WriteLine ("i is two.");
    break;
case 3 :
    Console.WriteLine ("i is three.");
    break;

    Console.WriteLine ("i is greater than 3.");
    break;
